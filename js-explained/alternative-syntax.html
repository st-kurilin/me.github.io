Alternative declarations


let a = 1;
var b = 1;      //almost the same as "let b = 1;"
const c = 1;    //declaring c as a constant that cannot be changed

a += 3;         //the same as "a = a + 3;"
a /= 2;         //the same as "a = a / 2;"
a++;            //almost the same as "a = a + 1;"
--a;            //almost the same as "a = a - 1;"
You really don't need spend time to get understanding difference between let and var. Spend it on something meaningfull!

There are alternative syntaxes for function declarations. 


const foo = (arg) => {
    return arg + 2;
};

const foo2 = arg => {
    return arg + 2;
};

const foo3 = arg => arg + 2;

function foo4(arg) {
    return arg + 2;
}
Note how brackets can be omitted for simple functions.

Alternative Data Structures

There is a special case for objects where all keys are numbers - arrays. 


let germanCars = ["BMV", "Audi"];
let frenchCars = ["Peugeot", "Citroen", "Renault"];
let danishCars = [];

let euCars = germanCars.concat(frenchCars).concat(danishCars);


//get total number
console.log(euCars.length); //-> 5
//add one more element
euCars.push("Fiat");
//access individual element
console.log(frenchCars[1]); //-> "Peugeot"
//change "Fiat" element
germanCars[1] = "volkswagen";
You can use objects in all cases where array might be used, but using arrays might be way more efficient.

Alternative Flow Control Structures

There is also alternative syntax for loops declarations 


//loop with condition
let c = 1;
while (c < 10) {
    console.log("C = ", c);
    c++;
}

//shorthand syntax usage with couter
for (let c = 1; c < 10; c++) {
    console.log("C = ", c);
}

And alternative syntax to check value 


//classical if-then-else
const a = "bar";
if (a == "foo") {
    console.log("a == foo");
} else if (a == "bar") {
    console.log("a != foo && a == bar");
} else {
    console.log("a != foo && a != bar");
}

//switch syntax
switch (a) {
    case "foo": {
        console.log("a == foo");
        break;
    }
    case "bar": {
        console.log("a != foo && a == bar");
        break;
    }
    default: console.log("a != foo && a != bar");
}
